#include <iostream>
using namespace std;
int main() {
    int a[]={121,11,115,1000};
    int i, dato;
    char band='F';
    
    dato=11;
    i=0;
    while((band=='F')&&(i<115)){
        if(a[i]==dato){
            band='V';
        }
        i++;
    }
    if(band=='F'){
        cout<<"El número a buscar no existe en el vector"<<endl;
    }
    else if(band=='V'){
        cout<<"El número está en la posición: "<<i<<endl;
    }

    return 0;
}
