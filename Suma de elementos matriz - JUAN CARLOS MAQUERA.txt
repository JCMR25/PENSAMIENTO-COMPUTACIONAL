#include <iostream>
using namespace std;
int main() {
    int n;
    cin>>n;
    int A[n][n];
    int suma=0;
    for (int i=0; i<n; i++) {
        for (int j= 0; j<n; j++) {
            cout<< "Elemento [" << i << "][" << j << "]: ";
            cin >> A[i][j];
            suma=suma+A[i][j];
        }
    }

    cout << "La suma de los elementos de la matriz es: " << suma << endl;

    return 0;
}
