#include <iostream>
using namespace std;

int main() {
    int x,y,z;
    cout << "Ingrese tres números: ";
    cin>>x>>y>>z;

    cout<<"Orden: ";

    if (x<y){
        if (x<z){
            if (y<z)
                cout<<x<<" "<<y<<" "<<z;
            else
                cout<<x<<" "<<z<<" "<<y;
        } else {
            cout<<z<<" "<<x<<" "<<y;
        }
    } else {
        if (x<z) {
            cout<<y<<" "<<x<<" "<<z;
        } else {
            if (y<z)
                cout<<y<<" "<<z<<" "<<x;
            else
                cout<<z<<" "<<y<<" "<<x;
        }
    }

    cout << endl;
    return 0;
}
