#include <iostream>
using namespace std;
int main() {
    int A[3][3];
    
    cout<<"Elementos de la matriz: "<<endl;
    for (int i=0; i<3; i++){
        for (int j=0; j<3; j++){
            cout<<"["<<i<<"]["<<j<<"]: ";
            cin>>A[i][j];
        }
    }
    
    int determinante=(A[0][0]*A[1][1]*A[2][2])+(A[0][1]*A[1][2]*A[2][0])+(A[0][2]*A[1][0]*A[2][1])-(A[0][2]*A[1][1]*A[2][0])-(A[0][0]*A[1][2]*A[2][1])-(A[0][1]*A[1][0]*A[2][2]);
    
    cout<<"La determinante de la matriz es: "<<determinante<<endl;
    
    return 0;
}
