#include <iostream>
using namespace std;

int main() {
    int n;
    cout<<"Ingrese el valor de n: ";
    cin>>n;

    int tabla[n][n];

    for (int i=0; i<n; i++) {
        for (int j=0; j<n; j++) {
            tabla[i][j] = (i + 1) * (j + 1);
        }
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
                cout << "   " << tabla[i][j];
        }
        cout << endl;
    }

    return 0;
}
