#include <iostream>
using namespace std;

int main() {
    int n;
    cout<<"Cantidad de datos: ";
    cin >> n;

    double A[n];
    double suma=0;

    for (int i=0; i<n; i++) {
        cin>>A[i];
        suma=suma+A[i];
    }

    double media=suma/n;
    cout<<"La media aritmética es: "<<media<<endl;

    return 0;
}
