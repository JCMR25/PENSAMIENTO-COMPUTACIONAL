#include <iostream>
using namespace std;

bool comparar(int A[4], int B[4]) {
    for (int i=0; i<4; i++) {
        if (A[i] != B[i]) {
            return false;
        }
    }
    return true;
}

int main() {
    int A[4] = {2,1,3,4};
    int B[4] = {2,2,8,4};

    if (comparar(A, B)) {
        cout<<"Los arrays son iguales."<<endl;
    } else {
        cout<<"Los arrays son diferentes."<<endl;
    }

    return 0;
}
